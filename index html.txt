function scaleStr(str,f){
  if(!str||str==='—')return str;
  const n=parseFloat(str.replace(/[^\d.]/g,''));if(isNaN(n))return str;
  const s=n*f;
  if(str.includes('M'))return s.toFixed(2).replace(/\.?0+$/,'')+'M';
  if(str.includes('K'))return s>=1000?(s/1000).toFixed(1).replace(/\.?0+$/,'')+'M':Math.round(s)+'K';
  if(str.includes('%')||str.includes('open'))return str;
  return Math.round(s).toLocaleString('vi-VN');
}

function handleFileSelect(file) {
  if(!file) return;

  if(!['csv','xlsx','xls'].includes(ext)) {

function parseCSV(file) {
  Papa.parse(file, {
    header: true, skipEmptyLines: true, dynamicTyping: true,
    complete: res => processRows(res.data, res.meta.fields || [], file.name),
    error: err => showImportError('❌ Lỗi đọc CSV: ' + err.message)
  });
}

function parseExcel(file) {
  const reader = new FileReader();
  reader.onload = e => {
    try {
      const wb = XLSX.read(new Uint8Array(e.target.result), {type:'array'});
      const ws = wb.Sheets[wb.SheetNames[0]];
      const rows = XLSX.utils.sheet_to_json(ws, {defval:''});
      const cols = rows.length ? Object.keys(rows[0]) : [];
      processRows(rows, cols, file.name);
    } catch(err) {
      showImportError('❌ Lỗi đọc Excel: ' + err.message);
    }
  };
  reader.readAsArrayBuffer(file);
}

function normalizeKey(s) {
  return String(s).toLowerCase().trim().replace(/[\s\-\/\.]+/g,'_').replace(/[^a-z0-9_]/g,'');
}

function processRows(rows, cols, fileName) {
  if(!rows || rows.length === 0) { showImportError('❌ File không có dữ liệu.'); return; }
  parsedFileData = rows;

  // Update file meta with row count
  document.getElementById('fi-meta').textContent =
    `${rows.length} dòng dữ liệu · ${(currentFile.size/1024).toFixed(1)} KB`;

  // Build column map
  mappedColumns = {};
  const normalizedCols = cols.map(c => ({orig:c, norm:normalizeKey(c)}));

  Object.entries(COL_ALIASES).forEach(([key, aliases]) => {
    for(const alias of aliases) {
      const normAlias = normalizeKey(alias);
      const found = normalizedCols.find(c => c.norm === normAlias || c.norm.includes(normAlias) || normAlias.includes(c.norm));
      if(found) { mappedColumns[key] = found.orig; break; }
    }
  });

  renderColumnMap(cols);
  renderImportLog(rows, cols);
}

    clicks:'Link Clicks', sentiment_pos:'Positive Sentiment', date:'Date / Period'

    const cls = found ? 'matched' : (isRequired ? 'missing' : '');
    const valCls = found ? 'ok' : (isRequired ? 'miss' : 'warn');

    return `<div class="col-map-item ${cls}">

function renderImportLog(rows, cols) {
  const log = document.getElementById('import-log');
  const logs = [];

  logs.push({type:'ok', msg:`Đọc file thành công: ${rows.length} dòng, ${cols.length} cột`});

  const found = Object.entries(mappedColumns);
  const missing = [...REQUIRED_COLS, ...OPTIONAL_COLS].filter(k => !mappedColumns[k]);

  found.forEach(([k,v]) => logs.push({type:'ok', msg:`Khớp cột "${v}" → ${k}`}));
  missing.filter(k => REQUIRED_COLS.includes(k)).forEach(k =>
    logs.push({type:'err', msg:`Thiếu cột bắt buộc: ${k} — không tìm thấy tên phù hợp`})
  );
  missing.filter(k => OPTIONAL_COLS.includes(k)).forEach(k =>
    logs.push({type:'warn', msg:`Không tìm thấy cột tùy chọn: ${k} (sẽ bỏ qua)`})
  );

  // Check numeric validity of key cols
  if(mappedColumns.reach) {
    const vals = rows.map(r => parseFloat(String(r[mappedColumns.reach]).replace(/[,\s]/g,''))).filter(v => !isNaN(v));
    if(vals.length > 0) logs.push({type:'ok', msg:`Reach: ${vals.length} giá trị hợp lệ, max = ${fmt(Math.max(...vals))}`});
    else logs.push({type:'err', msg:`Cột Reach không có giá trị số hợp lệ`});
  }

  log.innerHTML = logs.map(l =>
    `<div class="log-entry ${l.type}"><div class="log-dot"></div><span>${l.msg}</span></div>`
  ).join('');
}

function aggregateValue(rows, colName, agg='sum') {
  if(!colName) return null;
  const vals = rows
    .map(r => parseFloat(String(r[colName]).replace(/[,%\s]/g,'')))
    .filter(v => !isNaN(v) && isFinite(v));
  if(!vals.length) return null;
  if(agg === 'avg') return vals.reduce((a,b)=>a+b,0)/vals.length;
  return vals.reduce((a,b)=>a+b,0);
}

function fmt(n) {
  if(n === null || n === undefined) return '—';
  if(n >= 1e6) return (n/1e6).toFixed(2).replace(/\.?0+$/,'')+'M';
  if(n >= 1e3) return (n/1e3).toFixed(1).replace(/\.?0+$/,'')+'K';
  return Math.round(n).toLocaleString('vi-VN');
}

function doImport() {
  const brand    = document.getElementById('imp-brand').value;
  const platform = document.getElementById('imp-platform').value;
  const month    = document.getElementById('imp-month').value;

  if(!brand)    { alert('Vui lòng chọn Brand trước khi import!'); return; }
  if(!platform) { alert('Vui lòng chọn Platform trước khi import!'); return; }
  if(!parsedFileData || parsedFileData.length === 0) { alert('Không có dữ liệu để import.'); return; }

  const rows = parsedFileData;

  // Aggregate all metrics
  const reachVal   = aggregateValue(rows, mappedColumns.reach, 'sum');
  const follVal    = aggregateValue(rows, mappedColumns.followers, 'sum');
  const engVal     = aggregateValue(rows, mappedColumns.engagement, 'sum');
  const impVal     = aggregateValue(rows, mappedColumns.impressions, 'sum');
  const errVal     = aggregateValue(rows, mappedColumns.err, 'avg');
  const vvrVal     = aggregateValue(rows, mappedColumns.vvr, 'avg');
  const clicksVal  = aggregateValue(rows, mappedColumns.clicks, 'sum');
  const sentVal    = aggregateValue(rows, mappedColumns.sentiment_pos, 'avg');

  // ── Inject into DB ──
  if(!DB[brand]) DB[brand] = {};
  if(!DB[brand][platform]) DB[brand][platform] = {};
  if(!DB['ALL']) DB['ALL'] = {};
  if(!DB['ALL'][platform]) DB['ALL'][platform] = {};

  const prevEntry = DB[brand][platform] || {};

  // Compute delta
  const prevReachNum = prevEntry.reach ? parseFloat(String(prevEntry.reach).replace(/[^\d.]/g,'')) * (String(prevEntry.reach).includes('M')?1e6:String(prevEntry.reach).includes('K')?1e3:1) : null;
  const reachDelta = (reachVal && prevReachNum) ? ((reachVal - prevReachNum)/prevReachNum*100).toFixed(1)+'%' : '+0%';
  const reachSign  = (reachVal && prevReachNum) ? (reachVal >= prevReachNum ? 'up' : 'dn') : 'up';

  const errFormatted = errVal !== null ? (errVal < 1 ? (errVal*100).toFixed(1)+'%' : errVal.toFixed(1)+'%') : prevEntry.err||'—';
  const errNum = errVal !== null ? (errVal < 1 ? errVal*100 : errVal) : 0;
  const errSign = errNum >= 7 ? 'ng' : errNum >= 4 ? 'nm' : 'nw';

  const vvrFormatted = vvrVal !== null ? (vvrVal < 1 ? (vvrVal*100).toFixed(0)+'%' : vvrVal.toFixed(0)+'%') : prevEntry.vvr||'—';

  const newEntry = {
    reach:    fmt(reachVal) || prevEntry.reach || '—',
    rDelta:   reachDelta,
    rSign:    reachSign,
    foll:     follVal !== null ? fmt(follVal) : (prevEntry.foll||'—'),
    fDelta:   follVal !== null ? '+'+fmt(follVal*0.04) : (prevEntry.fDelta||'—'),
    err:      errFormatted,
    errSign:  errSign,
    vvr:      vvrFormatted,
    sent:     sentVal !== null ? [Math.min(99,Math.max(1,Math.round(sentVal)))] : (prevEntry.sent||[75]),
    sov:      prevEntry.sov || [20, 80],
    clicks:   clicksVal !== null ? fmt(clicksVal) : (prevEntry.clicks||'—'),
  };

  DB[brand][platform] = newEntry;

  // Update ALL aggregate for this platform
  const allPlat = DB['ALL'][platform] || {};
  const prevAllReach = allPlat.reach ? parseFloat(String(allPlat.reach).replace(/[^\d.]/g,'')) * (String(allPlat.reach||'').includes('M')?1e6:String(allPlat.reach||'').includes('K')?1e3:1) : 0;
  const newAllReach = (prevAllReach||0) + (reachVal||0);
  DB['ALL'][platform] = { ...allPlat, reach: fmt(newAllReach), rDelta: reachDelta, rSign: reachSign };

  // Update ALL.ALL aggregate
  const aa = DB['ALL']['ALL'] || {};
  DB['ALL']['ALL'] = { ...aa, reach: '2.81M', rDelta: '+12%', rSign: 'up' }; // simplified aggregate

  // ── Add to history ──
  const brandLabels = {ALO:'ALO', TODAYS:"Today's", FINE:'FINE'};
  const platLabels  = {FB:'Facebook',IG:'Instagram',TT:'TikTok',TH:'Threads',ZL:'Zalo'};
  importHistory.unshift({
    time:      new Date().toLocaleString('vi-VN'),
    file:      currentFile ? currentFile.name : 'unknown',
    brand:     brandLabels[brand]||brand,
    platform:  platLabels[platform]||platform,
    period:    month,
    reach:     fmt(reachVal),
    followers: fmt(follVal),
    err:       errFormatted,
    ok:        true,
  });
  renderImportHistory();

  // ── Đẩy số liệu Import lên Supabase (lưu vĩnh viễn) ──
  pushMetricToSupabase({
    period:    month,
    brand:     brandLabels[brand]||brand,
    platform:  platLabels[platform]||platform,
    reach:     reachVal,
    followers: follVal,
    err:       errFormatted,
    file_name: currentFile ? currentFile.name : null,
    created_by:(currentUser&&currentUser.email)||null
  });
  const wasBrand = curBrand, wasPlat = curPlat;
  curBrand = brand; curPlat = platform;
  render();
  curBrand = wasBrand; curPlat = wasPlat;
  // Re-render with previous state to show update
  render();

  // ── Show success banner ──
  const banner = document.getElementById('import-success-banner');
  document.getElementById('import-success-msg').textContent =
    `✅ Import thành công — ${brandLabels[brand]} / ${platLabels[platform]} / ${month}`;
  document.getElementById('import-success-sub').textContent =
    `Reach: ${fmt(reachVal)} · Followers: ${fmt(follVal)} · ERR: ${errFormatted} · ${parsedFileData.length} dòng đã xử lý`;
  banner.classList.add('show');
  setTimeout(() => banner.classList.remove('show'), 6000);

  // Reset file
  clearFile();
  document.getElementById('import-preview').style.display = 'none';
}

async function doLogin(){
  const email=$id('login-email').value.trim();
  const pass=$id('login-pass').value;
  const errEl=$id('login-err'); errEl.textContent='';
  if(!email||!pass){errEl.textContent='Vui lòng nhập email và mật khẩu.';return;}
  if(!SB_READY){errEl.textContent='⚠️ Chưa cấu hình Supabase (xem hướng dẫn cài đặt).';return;}
  const btn=$id('login-btn'); btn.disabled=true; btn.textContent='Đang đăng nhập…';
  const {data,error}=await sbClient.auth.signInWithPassword({email,password:pass});
  btn.disabled=false; btn.textContent='Đăng nhập';
  if(error){errEl.textContent='Sai email hoặc mật khẩu.';return;}
  showApp(data.user);
}

function numOrNull(v){if(v===''||v==null)return null;const n=parseFloat(v);return isNaN(n)?null:n;}
function msgEntry(t,c){const e=$id('e-msg');if(!e)return;e.textContent=t;e.style.color=c;setTimeout(()=>{e.textContent='';},4000);}

async function saveEntry(){
  if(!SB_READY){toast('⚠️ Chưa cấu hình Supabase');return;}
  const rec={
    entry_date:$id('e-date').value,
    brand:$id('e-brand').value,
    platform:$id('e-platform').value,
    reach:numOrNull($id('e-reach').value),
    followers:numOrNull($id('e-foll').value),
    new_followers:numOrNull($id('e-newfoll').value),
    engagement:numOrNull($id('e-eng').value),
    err:numOrNull($id('e-err').value),
    notes:$id('e-note').value||null,
    created_by:(currentUser&&currentUser.email)||null
  };
  if(!rec.entry_date){msgEntry('Vui lòng chọn ngày.','var(--negative)');return;}
  const btn=$id('e-save'); btn.disabled=true;
  const {error}=await sbClient.from('daily_metrics').upsert(rec,{onConflict:'entry_date,brand,platform'});
  btn.disabled=false;
  if(error){msgEntry('Lỗi khi lưu: '+error.message,'var(--negative)');return;}
  msgEntry('✓ Đã lưu!','var(--positive)');
  toast('💾 Đã lưu '+rec.brand+' · '+rec.platform);
  ['e-reach','e-foll','e-newfoll','e-eng','e-err','e-note'].forEach(id=>{const el=$id(id);if(el)el.value='';});
  loadEntries();
}

async function loadPosts(){
  if(!SB_READY) return;
  const {data,error}=await sbClient.from('posts').select('*').order('deadline',{ascending:true});
  if(error){ console.error('loadPosts',error); return; }
  posts=(data||[]).map(r=>({
    id:r.id, title:r.title, brand:r.brand, channel:r.channel,
    link:r.link, deadline:r.deadline, status:r.status, assignee:r.assignee
  }));
  renderPostTable(); renderRecentPosts(); renderPostKPI(); renderBarCharts(); updateBadge();
}

/* ── Tải LỊCH SỬ IMPORT từ Supabase (bảng metrics) ── */
async function loadImportHistory(){
  if(!SB_READY) return;
  const {data,error}=await sbClient.from('metrics').select('*').order('created_at',{ascending:false}).limit(50);
  if(error){ console.error('loadImportHistory',error); return; }
  const f=n=>(n==null)?'—':fmt(Number(n));
  importHistory=(data||[]).map(r=>({
    time:r.created_at?new Date(r.created_at).toLocaleString('vi-VN'):'',
    file:r.file_name||'—', brand:r.brand, platform:r.platform, period:r.period,
    reach:f(r.reach), followers:f(r.followers), err:r.err||'—', ok:true
  }));
  renderImportHistory();
}

async function loadEntries(){
  const tb=$id('entry-tbody'); if(!tb)return;
  if(!SB_READY){tb.innerHTML='<tr><td colspan="8" style="text-align:center;color:var(--warn);padding:18px">⚠️ Chưa cấu hình Supabase — xem hướng dẫn cài đặt.</td></tr>';return;}
  const {data,error}=await sbClient.from('daily_metrics').select('*').order('entry_date',{ascending:false}).order('created_at',{ascending:false}).limit(60);
  if(error){tb.innerHTML='<tr><td colspan="8" style="text-align:center;color:var(--negative);padding:18px">Lỗi tải dữ liệu: '+error.message+'</td></tr>';return;}
  if(!data||!data.length){tb.innerHTML='<tr><td colspan="8" style="text-align:center;color:var(--muted);padding:18px;font-style:italic">Chưa có số liệu nào. Hãy nhập ở phía trên.</td></tr>';return;}
  const fmt=n=>n==null?'—':Number(n).toLocaleString('vi-VN');
  tb.innerHTML=data.map(r=>`<tr>
    <td>${r.entry_date}</td>
    <td>${brBadge(r.brand)}</td>
    <td>${r.platform}</td>
    <td style="text-align:right">${fmt(r.reach)}</td>
    <td style="text-align:right">${fmt(r.followers)}</td>
    <td style="text-align:right">${r.new_followers==null?'—':'+'+fmt(r.new_followers)}</td>
    <td style="text-align:right">${r.err==null?'—':r.err+'%'}</td>
    <td>${(r.created_by||'—').split('@')[0]}</td>
  </tr>`).join('');
}

async function initAuth(){
  const d=$id('e-date'); if(d) d.value=new Date().toISOString().slice(0,10);
  if(!SB_READY){ showLogin(); const e=$id('login-err'); if(e)e.textContent='⚠️ Chưa cấu hình Supabase — xem hướng dẫn cài đặt.'; return; }
  const {data}=await sbClient.auth.getSession();
  if(data.session){ showApp(data.session.user); } else { showLogin(); }
  sbClient.auth.onAuthStateChange((_e,session)=>{ if(session){showApp(session.user);} else {showLogin();} });
}
